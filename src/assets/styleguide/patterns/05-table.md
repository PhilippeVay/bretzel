    @example
    <table>
        <thead>
            <tr>
                <th>Ville</th>
                <th>Pays</th>
            </tr>
        </thead>
        <tfoot>
            <tr>
                <td>Pied de tableau 1</td>
                <td>Pied de tableau 2</td>
            </tr>
        </tfoot>
        <tbody>
            <tr>
                <td>Strasbourg</td>
                <td>France</td>
            </tr>
            <tr>
                <td>Berlin</td>
                <td>Allemagne</td>
            </tr>
        </tbody>
    </table>
